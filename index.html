Matano-Meters is a national-scale, fault-tolerant ecosystem for crowd-sourcing, verifying, and mapping real-time fuel prices across all licensed retail stations in Uganda. It serves the entire socio-economic spectrum via three interfaces powered by a unified API backend and asynchronous event-driven architecture.

2.9 Core Triggers

-- Auto-update trust score when report status changes
CREATE OR REPLACE FUNCTION fn_update_user_trust()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.status = 'verified' THEN
        UPDATE users
        SET trust_score = LEAST(100, trust_score + 2),
            verified_count = verified_count + 1
        WHERE id = NEW.reporter_user_id;
    ELSIF NEW.status = 'rejected' THEN
        UPDATE users
        SET trust_score = GREATEST(0, trust_score - 5),
            rejected_count = rejected_count + 1
        WHERE id = NEW.reporter_user_id;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_price_report_update_trust
    AFTER UPDATE OF status ON price_reports
    FOR EACH ROW
    WHEN (OLD.status = 'pending'
          AND NEW.status IN ('verified', 'rejected'))
    EXECUTE FUNCTION fn_update_user_trust();

3.4 Core State Machine Engine

// Node.js 20+ or Go 1.22+
// Constraint: Total response < 4,000ms

const STATES = {
    WELCOME:           { menu: true,  ttl: 120 },
    SELECT_REGION:     { menu: true,  ttl: 120 },
    SELECT_DISTRICT:   { menu: true,  ttl: 120 },
    SELECT_SUBCOUNTY:  { menu: true,  ttl: 120 },
    SELECT_STATION:    { menu: true,  ttl: 120 },
    SELECT_FUEL:       { menu: true,  ttl: 120 },
    ENTER_PRICE:       { prompt: true, ttl: 120 },
    CONFIRM:           { menu: true,  ttl: 60  },
    VIEW_STATIONS:     { display: true, ttl: 120 },
    SUBMITTED:         { terminal: true, ttl: 30  },
    ERROR:             { terminal: true, ttl: 30  },
};

async function handleUSSDRequest(msisdn, sessionId, inputText) {
    const msisdnHash = crypto.createHash('sha256')
        .update(APP_SALT + msisdn).digest('hex');
    const sessionKey = `session:${msisdnHash}`;
    const lockToken = uuidv4();

    // Acquire distributed lock
    const locked = await redis.set(
        `session:${msisdnHash}:lock`, lockToken, 'NX', 'EX', 30);
    if (!locked) return composeResponse('END', 'Please try again.');

    try {
        let session = await redis.hgetall(sessionKey);
        if (!session || isEmpty(session)) {
            session = { state: 'WELCOME', sequence: 0,
                        channel: 'ussd', locale: detectLocale(msisdn) };
            await redis.hset(sessionKey, session);
            await redis.expire(sessionKey, STATES.WELCOME.ttl);
            return composeResponse('CON', formatWelcome(session.locale));
        }

        const input = inputText.trim();
        if (input === '0')
            return handleBackNavigation(session, sessionKey);
        if (input === '000') {
            session.state = 'WELCOME';
            await redis.hset(sessionKey, session);
            return composeResponse('CON', formatWelcome(session.locale));
        }

        let response;
        switch (session.state) {
            case 'WELCOME':
                response = await handleWelcome(session, sessionKey, input);
                break;
            case 'SELECT_REGION':
                response = await handleSelectRegion(session, sessionKey, input);
                break;
            case 'SELECT_DISTRICT':
                response = await handleSelectDistrict(session, sessionKey, input);
                break;
            case 'SELECT_STATION':
                response = await handleSelectStation(session, sessionKey, input);
                break;
            case 'SELECT_FUEL':
                response = await handleSelectFuel(session, sessionKey, input);
                break;
            case 'ENTER_PRICE':
                response = await handleEnterPrice(session, sessionKey, input);
                break;
            case 'CONFIRM':
                response = await handleConfirm(session, sessionKey, input);
                break;
            case 'VIEW_STATIONS':
                response = await handleViewStations(session, sessionKey, input);
                break;
            default:
                response = await handleError(session, sessionKey);
        }

        await redis.hset(sessionKey, 'sequence', ++session.sequence);
        await redis.expire(sessionKey,
            STATES[session.state]?.ttl ?? 120);

        kafka.produce('ussd.session.logs', {
            msisdn_hash: msisdnHash, session_id: sessionId,
            state: session.state, sequence: session.sequence,
            user_input: maskInput(session.state, input),
        });
        return response;
    } finally {
        await redis.eval(
            'if redis.call("get", KEYS[1]) == ARGV[1] ' +
            'then redis.call("del", KEYS[1]) end',
            1, `session:${msisdnHash}:lock`, lockToken);
    }
}

3.5 Price Entry & Confirmation Handlers

async function handleEnterPrice(session, sessionKey, input) {
    const price = parseFloat(input.replace(/\s/g, ''));
    if (isNaN(price) || price <= 0)
        return composeResponse('CON', formatText(session.locale, 'INVALID_PRICE'));

    const bounds = getPriceBounds(session.fuel_type);
    if (price < bounds.min || price > bounds.max)
        return composeResponse('CON', formatText(session.locale,
            'PRICE_OUT_OF_RANGE', { min: bounds.min, max: bounds.max }));

    // Check against 48h moving median
    const median = await getMovingMedian(session.station_id, session.fuel_type);
    if (median !== null) {
        const deviation = Math.abs(price - median) / median * 100;
        if (deviation > DEVIATION_THRESHOLD_PCT)
            session.anomaly_flagged = true;
    }

    session.reported_price = price;
    session.state = 'CONFIRM';
    await redis.hset(sessionKey, session);

    return composeResponse('CON', formatText(session.locale, 'CONFIRM_PROMPT',
        { station: session.station_code, fuel: session.fuel_type,
          price: price.toLocaleString() }));
}

async function handleConfirm(session, sessionKey, input) {
    if (input !== '1') {
        session.state = 'WELCOME';
        await redis.hset(sessionKey, session);
        return composeResponse('CON', formatText(session.locale, 'REPORT_CANCELLED'));
    }

    // Publish to Kafka for async verification
    await kafka.produce('price.reports', {
        station_id: session.station_id,
        reporter_msisdn_hash: sessionKey.replace('session:', ''),
        fuel_type: session.fuel_type,
        reported_price: session.reported_price,
        channel: 'ussd',
        anomaly_flagged: session.anomaly_flagged || false,
    });

    session.state = 'SUBMITTED';
    await redis.hset(sessionKey, session);
    return composeResponse('END', formatText(session.locale, 'REPORT_SUCCESS'));
}

async function handleViewStations(session, sessionKey, input) {
    const top = await getTopCheapestStations(
        session.region_id, session.district_id, 'petrol', 3);

    if (top.length === 0)
        return composeResponse('END', 'No stations found in this area.');

    const lines = top.map((s, i) =>
        `${i+1}. ${s.name} — UGX ${s.price.toLocaleString()}/L`);
    return composeResponse('END', 'TOP 3 CHEAPEST\n' + lines.join('\n'));
}

async function findNearestStations(geoJsonPoint, radiusMeters = 5000) {
    const result = await pool.query(`
        SELECT s.id, s.name, s.station_code, fo.name AS operator,
               vp.price, vp.fuel_type,
               ST_Distance(s.location,
                   ST_SetSRID(ST_MakePoint($1, $2), 4326)::geography
               ) AS distance_meters
        FROM stations s
        JOIN fuel_operators fo ON fo.id = s.operator_id
        JOIN verified_prices vp
            ON vp.station_id = s.id AND vp.effective_until IS NULL
        WHERE s.is_active = TRUE
          AND ST_DWithin(s.location,
              ST_SetSRID(ST_MakePoint($1, $2), 4326)::geography, $3)
        ORDER BY vp.price ASC, distance_meters ASC
        LIMIT 5
    `, [geoJsonPoint.coordinates[0], geoJsonPoint.coordinates[1], radiusMeters]);
    return result.rows;
}

async function checkRateLimit(msisdnHash, endpoint) {
    const config = RATE_LIMITS[endpoint];
    const windowKey = `ratelimit:${msisdnHash}:${endpoint}:${
        Math.floor(Date.now() / config.window_ms)}`;
    const count = await redis.incr(windowKey);
    if (count === 1) await redis.pexpire(windowKey, config.window_ms);
    if (count > config.max_requests) {
        const duration = getEscalatingBlock(msisdnHash, endpoint);
        await redis.set(`blocked:${msisdnHash}`, '1', 'EX', duration);
        return { allowed: false, reason: 'RATE_LIMITED', duration };
    }
    return { allowed: true };
}

async function processPriceReport(report) {
    const user = await getUser(report.reporter_user_id);
    const tierConfig = TIER_CONFIG[user.tier];

    if (tierConfig.canReportDirect) {
        const check = await verifyAgainstMedian(report);
        if (check.passed) {
            await publishVerifiedPrice(report);
            return { status: 'verified', method: 'high_trust_direct' };
        }
        return { status: 'rejected', reason: 'extreme_deviation' };
    }

    // Low-trust: need 3 matching reports in 15 min window
    let group = await findOpenConsensusGroup(
        report.station_id, report.fuel_type);
    if (!group) {
        group = await createConsensusGroup(report); // TTL 15 min
        return { status: 'pending', method: 'consensus_initiated' };
    }

    await addVoteToGroup(group.id, report);
    if (group.current_votes + 1 >= group.min_required_votes) {
        const price = calculateGroupMedian(group.id);
        await publishVerifiedPrice({ ...report, reported_price: price });
        return { status: 'verified', method: 'consensus_achieved' };
    }
    return { status: 'pending', method: 'consensus_waiting' };
}

4.5 Moving Median Anti-Cheat Engine

const MEDIAN_WINDOW_HOURS = 48;
const MAX_DEVIATION_PCT = 25;

async function computeMovingMedian(stationId, fuelType) {
    const cacheKey = `median:${stationId}:${fuelType}:48h`;
    const cached = await redis.get(cacheKey);
    if (cached) return JSON.parse(cached);

    const result = await db.query(`
        SELECT percentile_cont(0.5) WITHIN GROUP
                 (ORDER BY reported_price) AS median,
               percentile_cont(0.25) WITHIN GROUP
                 (ORDER BY reported_price) AS q1,
               COUNT(*) AS sample_count
        FROM price_reports
        WHERE station_id = $1 AND fuel_type = $2
          AND created_at >= NOW() - $3::INTERVAL
          AND status IN ('verified', 'pending')
    `, [stationId, fuelType, MEDIAN_WINDOW_HOURS + ' hours']);

    await redis.setex(cacheKey, 300, JSON.stringify(result.rows[0]));
    return result.rows[0];
}

async function verifyAgainstMedian(report) {
    const stats = await computeMovingMedian(
        report.station_id, report.fuel_type);
    if (!stats || stats.sample_count < 5)
        return { passed: true, confidence: 'low' };

    const deviation = Math.abs(report.reported_price - stats.median)
                      / stats.median * 100;
    if (deviation > MAX_DEVIATION_PCT)
        return { passed: false, reason: 'extreme_deviation',
                 deviation_pct: deviation };
    return { passed: true, deviation_pct: deviation };
}

Z-Score Time-Series Anomaly

async function detectPriceAnomaly(stationId, fuelType) {
    const prices = (await db.query(`
        SELECT reported_price FROM price_reports
        WHERE station_id = $1 AND fuel_type = $2
          AND created_at >= NOW() - INTERVAL '6 hours'
          AND status = 'verified'
        ORDER BY created_at DESC
    `, [stationId, fuelType])).rows.map(r => parseFloat(r.reported_price));

    if (prices.length < 3) return null;

    const mean = prices.reduce((a, b) => a + b) / prices.length;
    const stddev = Math.sqrt(
        prices.reduce((s, p) => s + Math.pow(p - mean, 2), 0) / prices.length);
    const zScore = Math.abs(prices[0] - mean) / (stddev || 1);

    if (zScore > 3.0) {
        await db.anomaly_alerts.create({
            alert_type: 'price_spike',
            severity: zScore > 4 ? 'critical' : 'warning',
            station_id: stationId, fuel_type: fuelType,
            reported_price: prices[0], median_baseline: mean,
            deviation_pct: Math.abs(prices[0] - mean) / mean * 100,
        });
        return { anomaly: true, zScore,
                 severity: zScore > 4 ? 'critical' : 'warning' };
    }
    return { anomaly: false, zScore };
}

4.6 Behavioral Detection

Inter-arrival Time Analysis

async function analyzeSubmissionPattern(msisdnHash) {
    const timestamps = await redis.lrange(
        `timestamps:${msisdnHash}`, 0, 49);
    if (timestamps.length < 10) return null;

    const intervals = [];
    for (let i = 1; i < timestamps.length; i++)
        intervals.push(timestamps[i-1] - timestamps[i]);

    const mean = intervals.reduce((a, b) => a + b) / intervals.length;
    const stddev = Math.sqrt(intervals.reduce(
        (s, i) => s + Math.pow(i - mean, 2), 0) / intervals.length);
    const cv = stddev / mean;  // Coefficient of variation

    // Human CV >= 0.5, Bot CV < 0.2
    if (cv < 0.2 && mean < 30000)
        return { pattern: 'automated', confidence: 'high',
                 coefficient_variation: cv };
    return { pattern: 'human_like', coefficient_variation: cv };
}

Geospatial DBSCAN Cluster Detection

async function detectGeospatialSybil(windowMinutes = 60) {
    const clusters = await db.query(`
        SELECT ST_ClusterDBSCAN(
                 location_lat || ',' || location_lng, 0.001, 5
               ) OVER () AS cluster_id,
               COUNT(*) AS report_count,
               COUNT(DISTINCT reporter_user_id) AS unique_reporters,
               ARRAY_AGG(DISTINCT reporter_user_id) AS reporter_ids
        FROM price_reports
        WHERE created_at >= NOW() - $1::INTERVAL
          AND location_lat IS NOT NULL
        GROUP BY cluster_id
        HAVING COUNT(DISTINCT reporter_user_id) >= 5
           AND COUNT(*) >= 10
    `, [windowMinutes + ' minutes']);

    for (const c of clusters.rows) {
        const scores = await getTrustScores(c.reporter_ids);
        const avg = scores.reduce((a, b) => a + b) / scores.length;
        if (avg < 20 && c.report_count > 20) {
            for (const uid of c.reporter_ids)
                await blacklistUser(uid, 'sybil_geo_cluster');
        }
    }
}

async function emergencyPriceFreeze(stationId, reason) {
    // Close current active price
    await db.verified_prices.update(
        { effective_until: NOW() },
        { where: { station_id: stationId, effective_until: null } }
    );

    const lastPrice = (await db.stations.findById(stationId)).last_known_price;
    await db.verified_prices.create({
        station_id: stationId, price: lastPrice,
        verification_method: 'emergency_freeze',
        effective_from: NOW(), effective_until: NOW() + INTERVAL '24 hours',
    });
    await notifyStationOperator(station.operator_id, {
        type: 'price_freeze', station_id: stationId,
        reason, frozen_price: lastPrice, duration: '24 hours',
    });
    await invalidateStationCache(stationId);
}
